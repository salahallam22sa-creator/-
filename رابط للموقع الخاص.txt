<script>
let data = JSON.parse(localStorage.getItem("data")) || {
teachers:[], activities:[], lessons:[], issues:[], thanks:[]
};

function save(){localStorage.setItem("data",JSON.stringify(data));}

function show(section){
document.querySelectorAll('.card').forEach(c=>c.style.display='none');
document.getElementById(section).style.display='block';
}

function add(type){
if(type==="teachers"){
data.teachers.push({
name:tName.value,score:tScore.value,note:tNote.value
});
}
if(type==="activities"){
data.activities.push({
title:aTitle.value,desc:aDesc.value
});
}
save();
render();
}

function render(){
teachersList.innerHTML="";
data.teachers.forEach(i=>{
teachersList.innerHTML+=`<li>${i.name} - ${i.score}</li>`;
});
}

function printPDF(){
window.print();
}

render();
</script>
